def tri_Insertion(T):
    n=len(T)
    if n<=1:
        return T
    for j in range(1,n):
        key=T[j]
        i=j
        while i>0 and key<T[i-1]:
            i=i-1
        #Insertion
        if i!=j:
            T[:j+1]=T[:i]+[key]+T[i:j]

def fusion(sorted1,sorted2):
    i,n1=0,len(sorted1)
    j,n2=0,len(sorted2)
    result,current=[],_
    while (i<n1 and j<n2):
        if (sorted1[i]<sorted2[j]):
            current=sorted1[i]
            i+=1
        else:
            current=sorted2[j]
            j+=1
        result.append(current)
    result.extend(sorted1[i:]+sorted2[j:])
    return result

def partition(T,p,r):
    if T==[]:
        return -1
    i=p-1
    x=T[r]
    for j in range(p,r):
        if T[j]<=x:
            i+=1
            T[i],T[j]=T[j],T[i]
    T[r],T[i+1]=T[i+1],T[r]
    return i+1

def quick_sort(T,p=None,r=None):
    if T==[]:
        return None
    if p is None:
        p=0
    if r is None:
        r=len(T)-1
    if p<r:
        d=partition(T,p,r)
        quick_sort(T,d+1,r)
        quick_sort(T,p,d-1)

def randomise_partition(T,p,r):
    if T==[]:

    i=randint(p,r)
    T[i],T[r]=T[r],T[i]
    return partition(T,p,r)

def counting_sort(T):
    if T ==[]:
        return None
    k=max(T)+1
    occ=[0]*k
    for i in T:
        occ[i]+=1
    j=0
    for k,o in enumerate(occ):
        for i in range(o):
            T[j]=k
            j+=1

def counting_sort2(T):
    n=len(T)
    k=max(T)+1
    occ=[0]*k
    B=[0]*(n+1)
    for t in T:
        occ[t]+=1
    for j in range(1,k):
        occ[j]+=occ[j-1]

    for j in range(n-1,-1,-1):
        B[occ[T[j]]]=T[j]
        occ[T[j]]-=1
    return B[1:]

def tri_denombrement( A ):
    n =len(A)
    B =[0]*(n+1)
    k=max(A)+1
    C=[0]*k
    for j in range ( n ):
    C [A[j]]=C[A[j]]+1
    for i in range (1,k):
    C [ i ]= C [ i ] + C [i -1]

    B [ C [ A [ j ]]]= A[ j ]
    C [ A [ j ]]= C [ A [j ]] -1
    return B [1:]

def centree_reduit(T):
    mn,mx=min(T),max(T)
    assert mn<mx
    B=[]
    for v in T:
        B.append((v-mn)/(mx+mn))
    return (B,mn,mx)
